57  riak_pb_message *pbresp,

60  riak_context *ctx = (riak_context*)(rev->context);
61  RpbErrorResp *errresp = rpb_error_resp__unpack(ctx->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
62  riak_error_response *response = (riak_error_response*)(ctx->malloc_fn)(sizeof(riak_error_response));
63  *done = RIAK_TRUE;
64  riak_free(ctx, &pbresp);
65  if (response == NULL) {
66  rpb_error_resp__free_unpacked(errresp, ctx->pb_allocator);
67  return ERIAK_OUT_OF_MEMORY;
68  }
69  response->_internal = errresp;
70  response->errcode = errresp->errcode;
71  response->errmsg = riak_binary_populate_from_pb(ctx, &(errresp->errmsg));
72  if (response->errmsg == NULL) {
73  riak_free(ctx, &response);
74  rpb_error_resp__free_unpacked(errresp, ctx->pb_allocator);
75  return ERIAK_OUT_OF_MEMORY;
76  }
77 
78  // Call user's error callback, if present
79  if (rev->error_cb) {
80  riak_response_callback cb = rev->error_cb;
81  (cb)(response, rev->cb_data);
82  }
83  *resp = response;
84 
85  return ERIAK_OK;
86 }

273 riak_error
274 riak_decode_get_response(riak_event *rev,
275  riak_pb_message *pbresp,
276  riak_get_response **resp,
277  riak_boolean_t *done) {
278  // decode the PB response etc
279  riak_context *ctx = (riak_context*)(rev->context);
280  RpbGetResp *rpbresp = rpb_get_resp__unpack(ctx->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
281  riak_log_debug(rev, "riak_decode_get_response len=%d/pb unpack = 0x%lx\n", pbresp->len, (long)(rpbresp));
282  *done = RIAK_TRUE;
283  if (rpbresp == NULL) {
284  return ERIAK_OUT_OF_MEMORY;
285  }
286  int i = 0;
287  riak_get_response *response = (riak_get_response*)(ctx->malloc_fn)(sizeof(riak_get_response));
288  if (response == NULL) {
289  return ERIAK_OUT_OF_MEMORY;
290  }
291  memset(response, '\0', sizeof(riak_get_response));
292  response->_internal = rpbresp;
293 
294  if (rpbresp->has_vclock) {
295  response->has_vclock = RIAK_TRUE;
296  response->vclock = riak_binary_populate_from_pb(ctx, &(rpbresp->vclock));
297  if (response->vclock == NULL) {
298  riak_free(ctx, &response);
299  return ERIAK_OUT_OF_MEMORY;
300  } }
301  if (rpbresp->has_unchanged) {
302  response->has_unmodified = RIAK_TRUE;
303  response->unmodified = rpbresp->unchanged;
304  }
305  if (rpbresp->n_content > 0) {
306  riak_error err = riak_object_new_array(ctx, &(response->content), rpbresp->n_content);
307  if (err != ERIAK_OK) {
308  riak_free(ctx, &response);
309  return 1;
310  }
311  response->n_content = rpbresp->n_content;
312  for(i = 0; i < rpbresp->n_content; i++) {
313  err = riak_object_new_from_pb(ctx, &(response->content[i]), rpbresp->content[i]);
314  // If any object allocation fails, clean up all previously allocated ones
315  if (err != ERIAK_OK) {
316  riak_object_free_array(ctx, &(response->content), i);
317  riak_free(ctx, &response);
318  return err;
319  }
320  }
321  }
322  *resp = response;
323 
324  return ERIAK_OK;
325 }

701  }

703  }

705  // of the old buckets to a new, larger array
706  // TODO: Geometric reallocation to minimize extra mallocs, maybe?
707  riak_uint32_t existing_buckets = response->n_buckets;
708  riak_uint32_t additional_buckets = listbucketresp->n_buckets;
709  if (response->buckets != NULL) {

725  ProtobufCBinaryData *binary = &(listbucketresp->buckets[i]);
726  response->buckets[i+existing_buckets] = riak_binary_new(ctx, binary->len, binary->data);
727  if (response->buckets[i+existing_buckets]->data == NULL) {
728  int j;
729  rpb_list_buckets_resp__free_unpacked(listbucketresp, ctx->pb_allocator);
730  for(j = 0; j < i; j++) {
731  riak_free(ctx, &(response->buckets[j+existing_buckets]));
732  }
733  riak_free(ctx, &(response->buckets));
734  riak_free(ctx, resp);
735 
736  return ERIAK_OUT_OF_MEMORY;
737  }
738  }

740  if (listbucketresp->has_done) {
741  riak_log_debug(rev, "%s", "HAS DONE");

743  }

747  riak_uint32_t existing_pbs = response->n_responses;

758  if (response->_internal == NULL) {
759  return ERIAK_OUT_OF_MEMORY;
760  }
