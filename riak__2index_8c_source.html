34 riak_error
35 riak_2index_request_encode(riak_operation *rop,
36  riak_binary *bucket,
37  riak_binary *index,
38  riak_2index_options *index_options,
39  riak_pb_message **req) {
40  riak_config *cfg = riak_operation_get_config(rop);
41  RpbIndexReq twoimsg = RPB_INDEX_REQ__INIT;
42 
43  riak_operation_set_bucket(rop, bucket);
44  riak_operation_set_index(rop, index);
45  riak_binary_copy_to_pb(&twoimsg.bucket, bucket);
46  riak_binary_copy_to_pb(&twoimsg.index, index);
47 
48  // process get options
49  if (index_options != NULL) {
50  twoimsg.qtype = index_options->qtype;
51  twoimsg.has_key = index_options->has_key;
52  if (index_options->has_key) {
53  riak_binary_copy_to_pb(&twoimsg.key, index_options->key);
54  }
55  twoimsg.has_range_min = index_options->has_range_min;
56  if (index_options->has_range_min) {
57  riak_binary_copy_to_pb(&twoimsg.range_min, index_options->range_min);
58  }
59  twoimsg.has_range_max = index_options->has_range_max;
60  if (index_options->has_range_max) {
61  riak_binary_copy_to_pb(&twoimsg.range_max, index_options->range_max);
62  }
63  twoimsg.has_return_terms = index_options->has_return_terms;
64  twoimsg.return_terms = index_options->return_terms;
65  twoimsg.has_stream = index_options->has_stream;
66  twoimsg.stream = index_options->stream;
67  twoimsg.has_max_results = index_options->has_max_results;
68  twoimsg.max_results = index_options->max_results;
69  twoimsg.has_continuation = index_options->has_continuation;
70  if (index_options->has_continuation) {
71  riak_binary_copy_to_pb(&twoimsg.continuation, index_options->continuation);
72  }
73  twoimsg.has_timeout = index_options->has_timeout;
74  twoimsg.timeout = index_options->timeout;
75  twoimsg.has_type = index_options->has_type;
76  if (index_options->has_type) {
77  riak_binary_copy_to_pb(&twoimsg.type, index_options->type);
78  }
79  twoimsg.has_term_regex = index_options->has_term_regex;
80  if (index_options->has_term_regex) {
81  riak_binary_copy_to_pb(&twoimsg.term_regex, index_options->term_regex);
82  }
83  twoimsg.has_pagination_sort = index_options->has_pagination_sort;
84  twoimsg.pagination_sort = index_options->pagination_sort;
85  }
86  riak_uint32_t msglen = rpb_index_req__get_packed_size(&twoimsg);
87  riak_uint8_t* msgbuf = (riak_uint8_t*)riak_config_clean_allocate(cfg, msglen);
88  if (msgbuf == NULL) {
89  return ERIAK_OUT_OF_MEMORY;
90  }
91  rpb_index_req__pack(&twoimsg, msgbuf);
92  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBINDEXREQ, msglen, msgbuf);
93  if (request == NULL) {
94  return ERIAK_OUT_OF_MEMORY;
95  }
96  *req = request;
97  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_2index_response_decode);
98 
99  return ERIAK_OK;
100 }

102 riak_error
103 riak_2index_response_decode(riak_operation *rop,
104  riak_pb_message *pbresp,
105  riak_2index_response **resp,
106  riak_boolean_t *done) {
107  // decode the PB response etc
108  riak_config *cfg = riak_operation_get_config(rop);
109  RpbIndexResp *rpbresp = rpb_index_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
110  if (rpbresp == NULL) {
111  return ERIAK_MESSAGE_FORMAT;
112  }
113 
114  // Initialize from an existing response
115  riak_2index_response *response = *resp;
116  // If this is NULL, there was no previous message
117  if (response == NULL) {
118  response = (riak_2index_response*)riak_config_clean_allocate(cfg, sizeof(riak_2index_response));
119  if (response == NULL) {
120  return ERIAK_OUT_OF_MEMORY;
121  }
122  }
123  *done = RIAK_FALSE;
124  if (rpbresp->has_done) {
125  *done = rpbresp->done;
126  }
127 
128  // Expand the vector of internal RpbListKeysResp links as necessary
129  riak_uint32_t existing_pbs = response->_n_responses;
130  if (existing_pbs > 0) {
131  if (riak_array_realloc(cfg,
132  (void***)&(response->_internal),
133  sizeof(RpbIndexResp*),
134  existing_pbs,
135  (existing_pbs+1)) == NULL) {
136  return ERIAK_OUT_OF_MEMORY;
137  }
138  } else {
139  response->_internal = (RpbIndexResp **)riak_config_allocate(cfg, sizeof(RpbIndexResp*));
140  if (response->_internal == NULL) {
141  return ERIAK_OUT_OF_MEMORY;
142  }
143  }
144  response->_internal[existing_pbs] = rpbresp;
145  response->_n_responses++;
146 
147  // If we are streaming, i.e., returning partial results or if we
148  // have the complete message, then assemble a user-consumable response
149  if (rop->streaming || *done) {
150  int i, j, chunk;
151  riak_uint32_t total_keys = 0;
152  riak_uint32_t total_results = 0;
153  for(i = 0; i < response->_n_responses; i++) {
154  total_keys += response->_internal[i]->n_keys;
155  total_results += response->_internal[i]->n_results;
156  }
157  response->keys = (riak_binary**)riak_config_clean_allocate(cfg, sizeof(riak_binary*) * total_keys);
158  if (response->keys == NULL) {
159  return ERIAK_OUT_OF_MEMORY;
160  }
161  response->n_keys = total_keys;
162  riak_error err = riak_pair_new_array(cfg , &(response->results), total_results);
163  if (err != ERIAK_OK) {
164  return err;
165  }
166  response->n_results = total_results;
167 
168  for(i = 0, j = 0, chunk = 0; i < response->n_keys; i++) {
169  RpbIndexResp *rpb_response = response->_internal[chunk];
170  response->keys[i] = riak_binary_copy_from_pb(cfg, &(rpb_response->keys[j]));
171  if (response->keys[i] == NULL) {
172  riak_free(cfg, &(response->keys));
173  return ERIAK_OUT_OF_MEMORY;
174  }
175  // If we have run out of keys in the current response,
176  // start working on the next response
177  if (++j >= response->_internal[chunk]->n_keys) {
178  chunk++;
179  j = 0;
180  }
181  }
182  for(i = 0, j = 0, chunk = 0; i < response->n_results; i++) {
183  RpbIndexResp *rpb_response = response->_internal[chunk];
184  err = riak_pairs_copy_from_pb(cfg, &(response->results[i]), rpb_response->results[j]);
185  if (err == ERIAK_OK) {
186  riak_free(cfg, &(response->results));
187  return ERIAK_OUT_OF_MEMORY;
188  }
189  // If we have run out of results in the current response,
190  // start working on the next response
191  if (++j >= response->_internal[chunk]->n_keys) {
192  chunk++;
193  j = 0;
194  }
195  }
196  }
197  // Is this correct? This would only be from the last message
198  RpbIndexResp *rpb_response = response->_internal[response->_n_responses-1];
199  response->has_continuation = rpb_response->has_continuation;
200  response->continuation = riak_binary_copy_from_pb(cfg, &(rpb_response->continuation));
201  response->has_done = rpb_response->has_done;
202  response->done = rpb_response->done;
203 
204  *resp = response;
205 
206  return ERIAK_OK;
207 }

209 riak_int32_t
210 riak_2index_response_print(riak_2index_response *response,
211  char **target,
212  riak_int32_t *len,
213  riak_int32_t *total) {
214  riak_int32_t wrote = 0;
215 
216  wrote += riak_print_int("n_keys", response->n_keys, target, len, total);
217  int i;
218  for(i = 0; (*len > 0) && (i < response->n_keys); i++) {
219  wrote += riak_print_binary("key", response->keys[i], target, len, total);
220  }
221  wrote += riak_print_int("n_results", response->n_results, target, len, total);
222  wrote += riak_pairs_print(response->results, response->n_results, target, len, total);
223  wrote += riak_print_bool("has_continuation", response->has_continuation, target, len, total);
224  wrote += riak_print_binary("continuation", response->continuation, target, len, total);
225  wrote += riak_print_bool("has_done", response->has_done, target, len, total);
226  wrote += riak_print_bool("done", response->done, target, len, total);
227 
228  return wrote;
229 }

231 void
232 riak_2index_response_free(riak_config *cfg,
233  riak_2index_response **resp) {
234  riak_2index_response *response = *resp;
235  if (response == NULL) {
236  return;
237  }
238  int i;
239  for(i = 0; i < response->n_keys; i++) {
240  riak_binary_free(cfg, &(response->keys[i]));
241  }
242  riak_free(cfg, &(response->keys));
243  riak_pairs_free(cfg, &(response->results), response->n_results);
244  riak_free(cfg, &(response->results));
245  for(i = 0; i < response->_n_responses; i++) {
246  rpb_index_resp__free_unpacked(response->_internal[i], cfg->pb_allocator);
247  }
248  riak_free(cfg, &(response->_internal));
249  riak_free(cfg, resp);
250 }
