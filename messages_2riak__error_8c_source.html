41  if (errresp == NULL) {
42  return ERIAK_MESSAGE_FORMAT;
43  }
44  riak_error_response *response = (riak_error_response*)(cfg->malloc_fn)(sizeof(riak_error_response));
45  *done = RIAK_TRUE;
46  riak_free(cfg, &pbresp);
47  if (response == NULL) {
48  rpb_error_resp__free_unpacked(errresp, cfg->pb_allocator);
49  return ERIAK_OUT_OF_MEMORY;
50  }
51  response->_internal = errresp;
52  response->errcode = errresp->errcode;
53  response->errmsg = riak_binary_copy_from_pb(cfg, &(errresp->errmsg));
54  if (response->errmsg == NULL) {
55  riak_free(cfg, &response);

61  if (rop->error_cb) {
62  riak_response_callback cb = rop->error_cb;
63  (cb)(response, rop->cb_data);
64  }
65  *resp = response;
66

68  &(rop->error),

70  response->errmsg);
71 
72  if (err) {
