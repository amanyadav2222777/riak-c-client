35 riak_delete_request_encode(riak_operation *rop,
36  riak_binary *bucket,
37  riak_binary *key,
38  riak_delete_options *options,
39  riak_pb_message **req) {
40 
41  riak_config *cfg = riak_operation_get_config(rop);
42  RpbDelReq delmsg = RPB_DEL_REQ__INIT;
43 
44  riak_binary_copy_to_pb(&delmsg.bucket, bucket);
45  riak_binary_copy_to_pb(&delmsg.key, key);
46 
47  // process delete options
48  if (options != NULL) {
49  if (options->has_dw) {
50  delmsg.has_dw = RIAK_TRUE;
51  delmsg.dw = options->dw;
52  }
53  if (options->has_n_val) {
54  delmsg.has_n_val = RIAK_TRUE;
55  delmsg.n_val = options->n_val;
56  }
57  if (options->has_pw) {
58  delmsg.has_pw = RIAK_TRUE;
59  delmsg.pw = options->pw;
60  }
61 
62  if (options->has_sloppy_quorum) {
63  delmsg.has_sloppy_quorum = RIAK_TRUE;
64  delmsg.sloppy_quorum = options->sloppy_quorum;
65  }
66  if (options->has_timeout) {
67  delmsg.has_timeout = RIAK_TRUE;
68  delmsg.timeout = options->timeout;
69  }
70  if (options->has_vclock) {
71  delmsg.has_vclock = RIAK_TRUE;
72  riak_binary_copy_to_pb(&delmsg.vclock, options->vclock);
73  }
74  if (options->has_w) {
75  delmsg.has_w = RIAK_TRUE;
76  delmsg.w = options->w;
77  }
78  }
79 
80  riak_uint32_t msglen = rpb_del_req__get_packed_size (&delmsg);
81  riak_uint8_t* msgbuf = (riak_uint8_t*)(cfg->malloc_fn)(msglen);
82  if (msgbuf == NULL) {
83  return ERIAK_OUT_OF_MEMORY;
84  }
85  rpb_del_req__pack (&delmsg, msgbuf);
86 
87  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBDELREQ, msglen, msgbuf);
88  if (request == NULL) {
89  return ERIAK_OUT_OF_MEMORY;
90  }
91  *req = request;
92  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_delete_response_decode);
93 
94  return ERIAK_OK;
95 }
