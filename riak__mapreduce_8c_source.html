35 riak_mapreduce_request_encode(riak_operation *rop,
36  riak_binary *content_type,
37  riak_binary *map_request,
38  riak_pb_message **req) {
39 
40  riak_config *cfg = riak_operation_get_config(rop);
41  RpbMapRedReq mapmsg = RPB_MAP_RED_REQ__INIT;
42 
43  riak_binary_copy_to_pb(&mapmsg.request, map_request);
44  riak_binary_copy_to_pb(&mapmsg.content_type, content_type);
45 
46  riak_uint32_t msglen = rpb_map_red_req__get_packed_size (&mapmsg);
47  riak_uint8_t* msgbuf = (riak_uint8_t*)(cfg->malloc_fn)(msglen);
48  if (msgbuf == NULL) {
49  return ERIAK_OUT_OF_MEMORY;
50  }
51  rpb_map_red_req__pack (&mapmsg, msgbuf);
52  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBMAPREDREQ, msglen, msgbuf);
53  if (request == NULL) {
54  return ERIAK_OUT_OF_MEMORY;
55  }
56  *req = request;
57  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_mapreduce_response_decode);
58 
59  return ERIAK_OK;
60 }

62 riak_error
63 riak_mapreduce_response_decode(riak_operation *rop,
64  riak_pb_message *pbresp,
65  riak_mapreduce_response **resp,
66  riak_boolean_t *done) {
67  // decode the PB response etc
68  riak_config *cfg = riak_operation_get_config(rop);
69  RpbMapRedResp *rpbresp = rpb_map_red_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
70  if (rpbresp == NULL) {
71  return ERIAK_MESSAGE_FORMAT;
72  }
73 
74  // Initialize from an existing response
75  riak_mapreduce_response *response = *resp;
76  // If this is NULL, there was no previous message
77  if (response == NULL) {
78  response = (riak_mapreduce_response*)riak_config_clean_allocate(cfg, sizeof(riak_mapreduce_response));
79  if (response == NULL) {
80  return ERIAK_OUT_OF_MEMORY;
81  }
82  }
83  *done = RIAK_FALSE;
84  if (rpbresp->has_done) {
85  riak_connection *cxn = riak_operation_get_connection(rop);
86  riak_log_debug(cxn, "%s", "HAS DONE");
87  *done = rpbresp->done;
88  }
89 
90  // Expand the vector of internal RpbListKeysResp links as necessary
91  riak_uint32_t existing_pbs = response->n_responses;
92  if (existing_pbs > 0) {
93  if (riak_array_realloc(cfg,
94  (void***)&(response->_internal),
95  sizeof(RpbMapRedResp*),
96  existing_pbs,
97  (existing_pbs+1)) == NULL) {
98  return ERIAK_OUT_OF_MEMORY;
99  }
100  } else {
101  response->_internal = (RpbMapRedResp **)riak_config_allocate(cfg, sizeof(RpbMapRedResp*));
102  if (response->_internal == NULL) {
103  return ERIAK_OUT_OF_MEMORY;
104  }
105  }
106  response->_internal[existing_pbs] = rpbresp;
107  response->n_responses++;
108 
109  // If we are streaming, i.e., returning partial results or if we
110  // have the complete message, then assemble a user-consumable response
111  if (rop->streaming || *done) {
112  int i;
113  response->msg = (riak_mapreduce_message**)riak_config_clean_allocate(cfg, sizeof(riak_mapreduce_message*) * response->n_responses);
114  if (response->msg == NULL) {
115  return ERIAK_OUT_OF_MEMORY;
116  }
117  for(i = 0; i < response->n_responses; i++) {
118  RpbMapRedResp *rpb_response = response->_internal[i];
119  response->msg[i] = (riak_mapreduce_message*)riak_config_clean_allocate(cfg, sizeof(riak_mapreduce_message));
120  if (response->msg[i] == NULL) {
121  riak_free(cfg, &(response->msg));
122  return ERIAK_OUT_OF_MEMORY;
123  }
124  riak_mapreduce_message *msg = response->msg[i];
125  msg->has_done = rpb_response->has_done;
126  msg->done = rpb_response->done;
127  msg->has_phase = rpb_response->has_phase;
128  msg->phase = rpb_response->phase;
129  msg->has_response = rpb_response->has_response;
130  msg->response = riak_binary_copy_from_pb(cfg, &(rpb_response->response));
131  }
132  }
133 
134  *resp = response;
135 
136  return ERIAK_OK;
137 }

139 riak_int32_t
140 riak_mapreduce_response_print(riak_mapreduce_response *response,
141  char **target,
142  riak_int32_t *len,
143  riak_int32_t *total) {
144  riak_int32_t wrote = 0;
145 
146  wrote += riak_print_int("n_responses", response->n_responses, target, len, total);
147  int i;
148  if (response->msg == NULL) {
149  return wrote;
150  }
151  for(i = 0; (*len > 0) && (i < response->n_responses); i++) {
152  riak_mapreduce_message *msg = response->msg[i];
153  wrote += riak_print_int("Response", i, target, len, total);
154  wrote += riak_print_bool("has_phase", msg->has_phase, target, len, total);
155  wrote += riak_print_int("phase", msg->phase, target, len, total);
156  wrote += riak_print_bool("has_response", msg->has_phase, target, len, total);
157  wrote += riak_print_binary("response", msg->response, target, len, total);
158  wrote += riak_print_bool("has_done", msg->has_phase, target, len, total);
159  wrote += riak_print_bool("done", msg->done, target, len, total);
160  }
161 
162  return wrote;
163 }

166 riak_mapreduce_response_free(riak_config *cfg,
167  riak_mapreduce_response **resp) {
168  riak_mapreduce_response *response = *resp;
169  int i;
170  for(i = 0; i < response->n_responses; i++) {
171  riak_binary_free(cfg, &(response->msg[i]->response));
172  riak_free(cfg, &(response->msg[i]));
173  rpb_map_red_resp__free_unpacked(response->_internal[i], cfg->pb_allocator);
174  }
175  riak_free(cfg, &(response->_internal));
176  riak_free(cfg, resp);
177 }
