29 riak_resolve_address(riak_context *ctx,
30  riak_addr_resolver resolver,
31  const char *host,
32  const char *portnum,
33  riak_addrinfo **addrinfo) {
34  riak_addrinfo addrhints;
35 
36  // Build the hints to tell getaddrinfo how to act.
37  memset(&addrhints, '\0', sizeof(riak_addrinfo));
38 // addrhints.ai_family = AF_UNSPEC; // IPv6 seems a bit broken for now
39  addrhints.ai_family = AF_INET; // IPv4 works like a champ
40  addrhints.ai_socktype = SOCK_STREAM;
41  addrhints.ai_protocol = IPPROTO_TCP; // We want a TCP socket
42  /* Only return addresses we can use. */
43  addrhints.ai_flags = EVUTIL_AI_ADDRCONFIG;
44 
45  // Use nice, platform agnostic DNS lookup and return an array of results
46  int err = resolver(host, portnum, &addrhints, addrinfo);
47  if (err != 0) {
48  riak_log_fatal_context(ctx, "Error while resolving '%s:%s': %s",
49  host, portnum, evutil_gai_strerror(err));
50  return ERIAK_DNS_RESOLUTION;
51  }
52 
53  return ERIAK_OK;
54 }

56 void
57 riak_print_host(riak_addrinfo *addrinfo,
58  char *target,
59  int len,
60  riak_uint16_t *port) {
61  struct sockaddr_in *ipv4;
62  struct sockaddr_in6 *ipv6;
63  switch (addrinfo->ai_addr->sa_family) {
64  case AF_INET:
65  ipv4 = (struct sockaddr_in*)addrinfo->ai_addr;
66  evutil_inet_ntop(ipv4->sin_family, &(ipv4->sin_addr), target, len);
67  *port = ntohs(ipv4->sin_port);
68  break;
69  case AF_INET6:
70  ipv6 = (struct sockaddr_in6*)addrinfo->ai_addr;
71  evutil_inet_ntop(ipv6->sin6_family, &(ipv6->sin6_addr), target, len);
72  *port = ntohs(ipv6->sin6_port);
73  break;
74  default:
75  riak_strlcpy(target, "<Unknown>", len);
76  port = 0;
77  }
78 }

80 riak_socket_t
81 riak_just_open_a_socket(riak_context *ctx,
82  riak_addrinfo *addrinfo) {
83 
84  riak_socket_t sock = socket(addrinfo->ai_family,
85  addrinfo->ai_socktype,
86  addrinfo->ai_protocol);
87  if (sock < 0) {
88  riak_log_fatal_context(ctx, "%s", "Could not just open a socket");
89  return -1;
90  }
91  int err = connect(sock, addrinfo->ai_addr, addrinfo->ai_addrlen);
92  if (err) {
93  // Since this is nonblocking, we'll need to treat some errors
94  // (like EINTR and EAGAIN) specially.
95  if (errno != EINPROGRESS && errno != EAGAIN) {
96  char ip[INET6_ADDRSTRLEN];
97  riak_uint16_t port;
98  riak_print_host(addrinfo, ip, sizeof(ip), &port);
99  EVUTIL_CLOSESOCKET(sock);
100  riak_log_fatal_context(ctx, "Could not connect a socket to host %s:%d [%s]\n", ip, port, strerror(errno));
101  return -1;
102  }
103  }
104  return sock;
105 }
