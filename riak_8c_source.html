61 static riak_error
62 riak_sync_request(riak_operation **rop_target,
63  void **response) {
64  riak_operation *rop = *rop_target;
65  riak_connection *cxn = riak_operation_get_connection(rop);
66  riak_operation_set_cb_data(rop, rop);
67 
68  riak_error err = riak_write(rop, riak_sync_write_cb, rop);
69  if (err) {
70  riak_log_critical(cxn, "%s", "Could not send request");
71  riak_operation_free(rop_target);
72  return err;
73  }
74 
75  riak_boolean_t done_streaming;
76  err = riak_read(rop, &done_streaming, riak_sync_read_cb, rop);
77 
78  *response = rop->response;
79  riak_operation_free(rop_target);
80  return err;
81 }

407 riak_error
408 riak_read(riak_operation *rop,
409  riak_boolean_t *done_streaming,
410  riak_io_cb read_cb,
411  void *read_cb_data) {
412  riak_connection *cxn = riak_operation_get_connection(rop);
413  riak_config *cfg = riak_connection_get_config(cxn);
414  riak_size_t buflen;
415  *done_streaming = RIAK_FALSE;
416 
417  while(RIAK_TRUE) {
418  // Are we in the middle of a message already?
419  if (rop->msglen_complete == RIAK_FALSE) {
420  // Read the first 32-bits which are the message size
421  // However, if a few size bytes were included during the last read, add them in
422  riak_uint32_t inmsglen = rop->msglen;
423  riak_size_t remaining_msg_len = sizeof(inmsglen) - rop->position;
424  riak_uint8_t *target = (riak_uint8_t*)(&inmsglen);
425  target += rop->position;
426  buflen = (read_cb)(read_cb_data, target, remaining_msg_len);
427  target = (riak_uint8_t*)(&inmsglen);
428  // If we can't ready any more bytes, stop trying
429  if (buflen != remaining_msg_len) {
430  riak_log_debug(cxn, "Expected %d bytes but received bytes = %d", remaining_msg_len, buflen);
431  if (buflen == 0) break;
432  // A few message size bytes of next message were in this buffer
433  // Stuff the partial size into rop->msglen and note the position
434  if (buflen < sizeof(inmsglen)) {
435  rop->position = buflen;
436  rop->msglen = inmsglen;
437  return ERIAK_OK;
438  }
439  return ERIAK_READ; // Something is hosed here
440  }
441 
442  rop->msglen_complete = RIAK_TRUE;
443  rop->msglen = ntohl(inmsglen);
444  riak_log_debug(cxn, "Read msglen = %d", rop->msglen);
445 
446  // TODO: Need to malloc new buffer each time?
447  rop->msgbuf = (riak_uint8_t*)riak_config_allocate(cfg, rop->msglen);
448  if (rop->msgbuf == NULL) {
449  riak_log_debug(cxn, "%s", "Could not allocate read buffer");
450  return ERIAK_READ;
451  }
452  } else {
453  riak_log_debug(cxn, "%s", "Continuation of partial message");
454  }
455 
456  riak_uint8_t *current_position = rop->msgbuf;
457  current_position += rop->position;
458  buflen = (read_cb)(read_cb_data, (void*)current_position, rop->msglen - rop->position);
459  riak_log_debug(cxn, "read %d bytes at position %d, msglen = %d", buflen, rop->position, rop->msglen);
460  rop->position += buflen;
461  // Are we done yet? If not, break out and wait for the next callback
462  if (rop->position < rop->msglen) {
463  riak_log_error(cxn, "%s","Partial message received");
464  return ERIAK_OK;
465  }
466  assert(rop->position == rop->msglen);
467 
468  riak_uint8_t msgid = (rop->msgbuf)[0];
469  riak_pb_message *pbresp = riak_pb_message_new(cfg, msgid, rop->msglen, rop->msgbuf);
470  riak_error result;
471  rop->position = 0; // Reset on success
472  rop->msglen = 0;
473  rop->msglen_complete = RIAK_FALSE;
474 
475  // Response varies by data type
476  riak_error_response *err_response = NULL;
477  // Assume we are doing a single loop, unless told otherwise
478  *done_streaming = RIAK_TRUE;
479  if (rop->decoder == NULL) {
480  riak_log_debug(cxn, "%d NOT IMPLEMENTED", msgid);
481  return ERIAK_READ;
482  }
483  if (msgid == MSG_RPBERRORRESP) {
484  result = riak_decode_error_response(rop, pbresp, &err_response, done_streaming);
485  // Convert error response to a null-terminated string
486  char errmsg[2048];
487  riak_binary_print(err_response->errmsg, errmsg, sizeof(errmsg));
488  riak_log_error(cxn, "ERR #%d - %s\n", err_response->errcode, errmsg);
489  if (rop->error_cb) {
490  (rop->error_cb)(err_response, rop->cb_data);
491  }
492  rop->response = NULL;
493  return ERIAK_SERVER_ERROR;
494  }
495  // Decode the message from Protocol Buffers
496  result = (rop->decoder)(rop, pbresp, &(rop->response), done_streaming);
497 
498  riak_free(cfg, &pbresp);
499  riak_free(cfg, &rop->msgbuf);
500 
501  // Something is amiss
502  if (result)
503  return ERIAK_READ;
504 
505  // Call the user-defined callback for this message, when finished
506  if (*done_streaming) {
507  if (rop->response_cb) {
508  (rop->response_cb)(rop->response, rop->cb_data);
509  }
510  break; // Done with current message
511  }
512  }
513 
514  return ERIAK_OK;
515 }

519 riak_error
520 riak_write(riak_operation *rop,
521  riak_io_cb write_cb,
522  void *write_cb_data) {
523  riak_pb_message *msg = rop->pb_request;
524  riak_uint8_t reqid = msg->msgid;
525  riak_uint8_t *msgbuf = msg->data;
526  riak_size_t len = msg->len;
527 
528  // Convert len to network byte order
529  riak_uint32_t msglen = htonl(len+1);
530  riak_int32_t wrote = (write_cb)(write_cb_data, (void*)&msglen, sizeof(msglen));
531  if (wrote == 0) return ERIAK_WRITE;
532  wrote = (write_cb)(write_cb_data, (void*)&reqid, sizeof(reqid));
533  if (wrote == 0) return ERIAK_WRITE;
534  if (len > 0) {
535  wrote = (write_cb)(write_cb_data, (void*)msgbuf, len);
536  if (wrote == 0) return ERIAK_WRITE;
537  }
538 #ifdef _RIAK_DEBUG
539  riak_connection *cxn = riak_operation_get_connection(rop);
540  riak_log_debug(cxn, "Wrote %d bytes", (int)len);
541  char buffer[10240];
542  riak_size_t buflen = sizeof(buffer);
543  char *pos = buffer;
544  wrote = 0;
545  int i;
546  for(i = 0; buflen > 0 && i < len; i++) {
547  wrote = snprintf(pos, buflen, "%02x", msgbuf[i]);
548  pos += wrote;
549  buflen -= wrote;
550  }
551  fprintf(stdout, "\n");
552  for(i = 0; buflen > 0 && i < len; i++) {
553  char c = '.';
554  if (msgbuf[i] > 31 && msgbuf[i] < 128) {
555  c = msgbuf[i];
556  }
557  wrote = snprintf(pos, buflen, "%c", c);
558  pos += wrote;
559  buflen -= wrote;
560  }
561  riak_log_debug(cxn, "%s", buffer);
562 #endif
563  return ERIAK_OK;
564 }
