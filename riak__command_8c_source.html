102 void
103 build_options(riak_command *commands,
104  struct option **long_options,
105  int *operation,
106  char *optstring) {
107  riak_int32_t num_commands = 0;
108  // Count the number of commands
109  for(; commands[num_commands].name != NULL; num_commands++);
110  struct option *opts = (struct option*)calloc(sizeof(struct option), num_commands+1);
111  if (opts == NULL) {
112  fprintf(stderr, "Could not allocate options array!\n");
113  exit(1);
114  }
115 
116  int i;
117  for(i = 0; i < num_commands; i++) {
118  // Does this relate to an operation?
119  if (commands[i].argument == NULL) {
120  opts[i].name = commands[i].name;
121  opts[i].has_arg = no_argument;
122  opts[i].flag = operation;
123  opts[i].val = commands[i].val;
124  } else {
125  opts[i].name = commands[i].name;
126  if (commands[i].argument[0] != '\0') {
127  opts[i].has_arg = required_argument;
128  *optstring++ = commands[i].val;
129  *optstring++ = ':';
130  } else {
131  // Flag has no argument, but operation either
132  opts[i].has_arg = no_argument;
133  *optstring++ = commands[i].val;
134  }
135  opts[i].flag = NULL;
136  opts[i].val = commands[i].val;
137  }
138  }
139  *long_options = opts;
140 }

142 int
143 argument_check(FILE *fp,
144  const char *progname,
145  riak_command *commands,
146  riak_args *args) {
147  int i;
148  riak_boolean_t ok = RIAK_TRUE;
149  for(i = 0; commands[i].name != NULL; i++) {
150  if (args->operation == commands[i].val) {
151  if (commands[i].needs_bucket && args->has_bucket != RIAK_TRUE) {
152  fprintf(fp, "--bucket parameter required\n");
153  ok = RIAK_FALSE;
154  }
155  if (commands[i].needs_key && args->has_key != RIAK_TRUE) {
156  fprintf(fp, "--key parameter required\n");
157  ok = RIAK_FALSE;
158  }
159  if (commands[i].needs_value && args->has_value != RIAK_TRUE) {
160  fprintf(fp, "--value parameter required\n");
161  ok = RIAK_FALSE;
162  }
163  break;
164  }
165  }
166 
167  // Case of not found
168  if (commands[i].name == NULL) ok = RIAK_FALSE;
169 
170  if (!ok) {
171  return -1;
172  }
173  return 0;
174 }

177 int
178 riak_parse_args(int argc,
179  char *argv[],
180  riak_args *args) {
181 
182  static int operation; // static required for getopt
183  int c;
184 
185  memset((void*)args, '\0', sizeof(riak_args));
186  args->operation = MSG_RPBERRORRESP;
187  args->async = RIAK_FALSE;
188  args->iterate = 1;
189  args->port = 10017;
190  args->timeout = 10;
191  args->has_bucket = RIAK_FALSE;
192  args->has_key = RIAK_FALSE;
193  args->has_value = RIAK_FALSE;
194 
195  strcpy(args->host, "localhost");
196  strcpy(args->portnum, "10017");
197 
198  char optstring[128];
199  struct option *long_options = NULL;
200  build_options(s_commands, &long_options, &operation, optstring);
201 
202  // Keep parsing args as long as we can
203  while (1) {
204  /* getopt_long stores the option index here. */
205  int option_index = 0;
206 
207  c = getopt_long (argc, argv, optstring,
208  long_options, &option_index);
209 
210  // Detect the end of the options.
211  if (c == -1)
212  break;
213 
214  switch (c) {
215  case 0:
216  /* If this option set a flag, do nothing else now. */
217  if (long_options[option_index].flag != 0) {
218  break;
219  }
220  printf ("option %s", long_options[option_index].name);
221  if (optarg)
222  printf (" with arg %s", optarg);
223  printf ("\n");
224  break;
225 
226  case 'a':
227  args->async = RIAK_TRUE;
228  break;
229 
230  case 'b':
231  printf ("option -b with value `%s'\n", optarg);
232  riak_strlcpy(args->bucket, optarg, sizeof(args->bucket));
233  args->has_bucket = RIAK_TRUE;
234  break;
235 
236  case 'h':
237  printf ("option -h with value `%s'\n", optarg);
238  riak_strlcpy(args->host, optarg, sizeof(args->host));
239  break;
240 
241  case 'i':
242  printf ("option -i with value `%s'\n", optarg);
243  args->iterate = atol(optarg);
244  break;
245 
246  case 'k':
247  printf ("option -k with value `%s'\n", optarg);
248  riak_strlcpy(args->key, optarg, sizeof(args->key));
249  args->has_key = RIAK_TRUE;
250  break;
251 
252  case 'p':
253  printf ("option -p with value `%s'\n", optarg);
254  riak_strlcpy(args->portnum, optarg, sizeof(args->portnum));
255  args->port = atol(optarg);
256  break;
257 
258  case 't':
259  printf ("option -t with value `%s'\n", optarg);
260  args->timeout = atol(optarg);
261  break;
262 
263  case 'v':
264  printf ("option -v with value `%s'\n", optarg);
265  riak_strlcpy(args->value, optarg, sizeof(args->value));
266  args->has_value = RIAK_TRUE;
267  break;
268 
269  case '?':
270  /* getopt_long already printed an error message. */
271  operation = -1;
272  break;
273 
274  default:
275  return -1;
276  }
277  }
278  args->operation = operation;
279  if (argument_check(stderr, argv[0], s_commands, args) < 0) {
280  operation = -1;
281  }
282  free(long_options);
283  return operation;
284 }
