35 riak_error
36 riak_search_request_encode(riak_operation *rop,
37  riak_binary *bucket,
38  riak_binary *query,
39  riak_search_options *search_options,
40  riak_pb_message **req) {
41  riak_config *cfg = riak_operation_get_config(rop);
42  RpbSearchQueryReq srchmsg = RPB_SEARCH_QUERY_REQ__INIT;
43 
44  riak_operation_set_bucket(rop, bucket);
45  riak_binary_copy_to_pb(&srchmsg.q, query);
46  riak_binary_copy_to_pb(&srchmsg.index, bucket);
47 
48  // process get options
49  if(search_options != NULL) {
50  srchmsg.has_rows = search_options->has_rows;
51  srchmsg.rows = search_options->rows;
52  srchmsg.has_start = search_options->has_start;
53  srchmsg.start = search_options->start;
54  if (search_options->has_sort) {
55  srchmsg.has_sort = search_options->has_sort;
56  riak_binary_copy_to_pb(&srchmsg.sort, search_options->sort);
57  }
58  if (search_options->has_filter) {
59  srchmsg.has_filter = search_options->has_filter;
60  riak_binary_copy_to_pb(&srchmsg.filter, search_options->filter);
61  }
62  if (search_options->has_df) {
63  srchmsg.has_df = search_options->has_df;
64  riak_binary_copy_to_pb(&srchmsg.df, search_options->df);
65  }
66  if (search_options->has_op) {
67  srchmsg.has_op = search_options->has_op;
68  riak_binary_copy_to_pb(&srchmsg.op, search_options->op);
69  }
70  if (search_options->has_presort) {
71  srchmsg.has_presort = search_options->has_presort;
72  riak_binary_copy_to_pb(&srchmsg.presort, search_options->presort);
73  }
74  if (search_options->n_fl > 0) {
75  srchmsg.n_fl = search_options->n_fl;
76  srchmsg.fl = (ProtobufCBinaryData*)riak_config_clean_allocate(cfg, sizeof(ProtobufCBinaryData) * search_options->n_fl);
77  if (srchmsg.fl == NULL) {
78  return ERIAK_OUT_OF_MEMORY;
79  }
80  for(int i = 0; i < search_options->n_fl; i++) {
81  riak_binary_copy_to_pb(&(srchmsg.fl[i]), search_options->fl[i]);
82  }
83  }
84  }
85  riak_uint32_t msglen = rpb_search_query_req__get_packed_size (&srchmsg);
86  riak_uint8_t* msgbuf = (riak_uint8_t*)riak_config_clean_allocate(cfg, msglen);
87  if (msgbuf == NULL) {
88  return ERIAK_OUT_OF_MEMORY;
89  }
90  rpb_search_query_req__pack (&srchmsg, msgbuf);
91  riak_free(cfg, &srchmsg.fl);
92  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBSEARCHQUERYREQ, msglen, msgbuf);
93  if (request == NULL) {
94  return ERIAK_OUT_OF_MEMORY;
95  }
96  *req = request;
97  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_search_response_decode);
98 
99  return ERIAK_OK;
100 }

102 riak_error
103 riak_search_response_decode(riak_operation *rop,
104  riak_pb_message *pbresp,
105  riak_search_response **resp,
106  riak_boolean_t *done) {
107  // decode the PB response etc
108  riak_config *cfg = riak_operation_get_config(rop);
109  RpbSearchQueryResp *rpbresp = rpb_search_query_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
110  if (rpbresp == NULL) {
111  return ERIAK_MESSAGE_FORMAT;
112  }
113 
114  // Initialize from an existing response
115  riak_search_response *response = *resp;
116  // If this is NULL, there was no previous message
117  if (response == NULL) {
118  response = (riak_search_response*)riak_config_clean_allocate(cfg, sizeof(riak_search_response));
119  if (response == NULL) {
120  return ERIAK_OUT_OF_MEMORY;
121  }
122  }
123  *done = RIAK_TRUE;
124 
125  response->has_max_score = rpbresp->has_max_score;
126  response->max_score = rpbresp->max_score;
127  response->has_num_found = rpbresp->has_num_found;
128  response->num_found = rpbresp->num_found;
129  response->n_docs = rpbresp->n_docs;
130  if (rpbresp->n_docs > 0) {
131  response->docs = (riak_search_doc*)riak_config_clean_allocate(cfg, sizeof(riak_search_doc) * rpbresp->n_docs);
132  if (response->docs == NULL) {
133  riak_free(cfg, &response);
134  return ERIAK_OUT_OF_MEMORY;
135  }
136  for(int i = 0; i < rpbresp->n_docs; i++) {
137  riak_error err = riak_pairs_copy_array_from_pb(cfg, &(response->docs[i].fields), rpbresp->docs[i]->fields, rpbresp->docs[i]->n_fields);
138  if (err) {
139  riak_free(cfg, &(response->docs));
140  riak_free(cfg, &response);
141  return err;
142  }
143  response->docs[i].n_fields = rpbresp->docs[i]->n_fields;
144  }
145  }
146  response->_internal = rpbresp;
147 
148  *resp = response;
149 
150  return ERIAK_OK;
151 }

153 riak_int32_t
154 riak_search_response_print(riak_search_response *response,
155  char **target,
156  riak_int32_t *len,
157  riak_int32_t *total) {
158  riak_int32_t wrote = 0;
159 
160  wrote += riak_print_bool("has_max_score", response->has_max_score, target, len, total);
161  wrote += riak_print_float("max_score", response->max_score, target, len, total);
162  wrote += riak_print_bool("has_num_found", response->has_num_found, target, len, total);
163  wrote += riak_print_int("num_found", response->num_found, target, len, total);
164  wrote += riak_print_int("n_docs", response->n_docs, target, len, total);
165  if (response->docs == NULL) {
166  return wrote;
167  }
168  for(int i = 0; (*len > 0) && (i < response->n_docs); i++) {
169  wrote += riak_print_int("Document", i, target, len, total);
170  wrote += riak_print_int("n_fields", response->docs[i].n_fields, target, len, total);
171  wrote += riak_pairs_print(response->docs[i].fields,response->docs[i].n_fields, target, len, total);
172  }
173 
174  return wrote;
175 }
