29 riak_binary*
30 riak_binary_new(riak_config *cfg,
31  riak_size_t len,
32  riak_uint8_t *data) {
33  riak_binary *b = riak_config_allocate(cfg, sizeof(riak_binary));
34  // In the degenerate case, force the length to be zero
35  if (data == NULL) {
36  len = 0;
37  }
38  if (b) {
39  b->len = len;
40  b->data = riak_config_allocate(cfg, len);
41  b->managed = RIAK_TRUE;
42  if (b->data) {
43  memcpy((void*)b->data, (void*)data, len);
44  } else {
45  riak_free(cfg, &b);
46  }
47  }
48  return b;
49 }

159  target[i] = c;

167 riak_size_t
168 riak_binary_hex_print(riak_binary *bin,
169  char *target,
170  riak_uint32_t len) {
171  riak_size_t count = 0;
172  static char hex[] = "0123456789abcdef";
173  if (bin != NULL) {
174  for( ; count < bin->len && (count*2) < len-1; count++) {
175 
176  int nibble = (bin->data[count] & 0xf0) >> 4;
177  target[count*2] = hex[nibble];
178  nibble = (bin->data[count] & 0x0f);
179  target[count*2+1] = hex[nibble];
180  }
181  }
182  if (len > 0) target[count*2] = '\0';
183  return count*2;
184 }
