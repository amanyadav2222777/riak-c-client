164  char *target,

166  int i = 0;
167  for( ; (bin) && (i < (bin->len)) && i < (len-1); i++) {
168  char c = '.';
169  // Non-printable characters are replaced by a dot
170  if (bin->data[i] >= 32) c = bin->data[i];
171  target[i] = c;
172  }
173  if (len > 0) {
174  target[i] = '\0';
175  }
176  return i;

179 int
180 riak_binary_hex_print(riak_binary *bin,
181  char *target,
182  riak_uint32_t len) {
183  int count = 0;
184  static char hex[] = "0123456789abcdef";
185  if (bin != NULL) {
186  for( ; count < bin->len && (count*2) < len-1; count++) {
187 
188  int nibble = (bin->data[count] & 0xf0) >> 4;
189  target[count*2] = hex[nibble];
190  nibble = (bin->data[count] & 0x0f);
191  target[count*2+1] = hex[nibble];
192  }
193  }
194  if (len > 0) target[count*2] = '\0';
195  return count*2;
196 }
