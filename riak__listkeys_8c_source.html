41  riak_binary_copy_to_pb(&(listkeysreq.bucket), bucket);

65 riak_error
66 riak_listkeys_response_decode(riak_operation *rop,
67  riak_pb_message *pbresp,
68  riak_listkeys_response **resp,
69  riak_boolean_t *done) {
70  riak_config *cfg = riak_operation_get_config(rop);
71  RpbListKeysResp *listkeyresp = rpb_list_keys_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
72  if (listkeyresp == NULL) {
73  return ERIAK_MESSAGE_FORMAT;
74  }
75  int i;
76  // Initialize from an existing response
77  riak_listkeys_response *response = *resp;
78  // If this is NULL, there was no previous message
79  if (response == NULL) {
80  response = (cfg->malloc_fn)(sizeof(riak_listkeys_response));
81  if (response == NULL) {
82  return ERIAK_OUT_OF_MEMORY;
83  }
84  memset((void*)response, '\0', sizeof(riak_listkeys_response));
85  }
86  // Existing keys need some expansion, so copy the pointers
87  // of the old keys to a new, larger array
88  // TODO: Geometric reallocation to minimize extra mallocs, maybe?
89  riak_uint32_t existing_keys = response->n_keys;
90  riak_uint32_t additional_keys = listkeyresp->n_keys;
91  if (response->keys != NULL) {
92  if (riak_array_realloc(cfg,
93  (void***)&(response->keys),
94  sizeof(riak_binary*),
95  existing_keys,
96  (additional_keys+existing_keys)) == NULL) {
97  return ERIAK_OUT_OF_MEMORY;
98  }
99  } else {
100  response->keys = (riak_binary**)(cfg->malloc_fn)(sizeof(riak_binary*)*additional_keys);
101  if (response->keys == NULL) {
102  return ERIAK_OUT_OF_MEMORY;
103  }
104  }
105  response->n_keys += additional_keys;
106  for(i = 0; i < additional_keys; i++) {
107  ProtobufCBinaryData *binary = &(listkeyresp->keys[i]);
108  response->keys[i+existing_keys] = riak_binary_new(cfg, binary->len, binary->data);
109  if (response->keys[i+existing_keys]->data == NULL) {
110  int j;
111  rpb_list_keys_resp__free_unpacked(listkeyresp, cfg->pb_allocator);
112  for(j = 0; j < i; j++) {
113  riak_free(cfg, &(response->keys[j+existing_keys]));
114  }
115  riak_free(cfg, &(response->keys));
116  riak_free(cfg, resp);
117 
118  return ERIAK_OUT_OF_MEMORY;
119  }
120  }
121  response->done = RIAK_FALSE;
122  if (listkeyresp->has_done) {
123  riak_connection *cxn = riak_operation_get_connection(rop);
124  riak_log_debug(cxn, "%s", "HAS DONE");
125  response->done = listkeyresp->done;
126  }
127  *done = response->done;
128 
129  // Expand the vector of internal RpbListKeysResp links as necessary
130  riak_uint32_t existing_pbs = response->n_responses;
131  if (existing_pbs > 0) {
132  if (riak_array_realloc(cfg,
133  (void***)&(response->_internal),
134  sizeof(RpbListKeysResp*),
135  existing_pbs,
136  (existing_pbs+1)) == NULL) {
137  return ERIAK_OUT_OF_MEMORY;
138  }
139  } else {
140  response->_internal = (RpbListKeysResp **)(cfg->malloc_fn)(sizeof(RpbListKeysResp*));
141  if (response->_internal == NULL) {
142  return ERIAK_OUT_OF_MEMORY;
143  }
144  }
145  response->_internal[existing_pbs] = listkeyresp;
146  response->n_responses++;
147  *resp = response;
148 
149  return ERIAK_OK;
150 }

152 void
153 riak_listkeys_response_print(riak_listkeys_response *response,
154  char *target,
155  riak_size_t len) {
156  int i;
157  riak_size_t wrote = 0;
158  riak_int32_t left_to_write = len;
159  char name[2048];
160  if (left_to_write > 0) {
161  wrote = snprintf(target, left_to_write, "n_keys = %d\n", response->n_keys);
162  left_to_write -= wrote;
163  target += wrote;
164  }
165  for(i = 0; (left_to_write > 0) && (i < response->n_keys); i++) {
166  riak_binary_print(response->keys[i], name, sizeof(name));
167  wrote = snprintf(target, left_to_write, "%d - %s\n", i, name);
168  left_to_write -= wrote;
169  target += wrote;
170  }
171  if (left_to_write > 0) {
172  snprintf(target, left_to_write, "done = %d", response->done);
173  }
174 }
