33 riak_error
34 riak_put_request_encode(riak_operation *rop,
35  riak_object *riak_obj,
36  riak_put_options *options,
37  riak_pb_message **req) {
38 
39  riak_config *cfg = riak_operation_get_config(rop);
40  RpbPutReq putmsg = RPB_PUT_REQ__INIT;
41 
42  riak_binary_copy_to_pb(&(putmsg.bucket), riak_obj->bucket);
43 
44  // Is the Key provided?
45  if (riak_obj->has_key) {
46  putmsg.has_key = RIAK_TRUE;
47  riak_binary_copy_to_pb(&(putmsg.key), riak_obj->key);
48  }
49 
50  // Data content payload
51  RpbContent content;
52  riak_object_to_pb_copy(cfg, &content, riak_obj);
53  putmsg.content = &content;
54 
55  // process put options
56  if (options != NULL) {
57  if (options->has_asis) {
58  putmsg.has_asis = RIAK_TRUE;
59  putmsg.asis = options->asis;
60  }
61  if (options->has_dw) {
62  putmsg.has_dw = RIAK_TRUE;
63  putmsg.dw = options->dw;
64  }
65  if (options->has_if_none_match) {
66  putmsg.has_if_none_match = RIAK_TRUE;
67  putmsg.has_if_none_match = options->has_if_none_match;
68  }
69  if (options->has_if_not_modified) {
70  putmsg.has_if_not_modified = RIAK_TRUE;
71  putmsg.has_if_not_modified = options->has_if_not_modified;
72  }
73  if (options->has_n_val) {
74  putmsg.has_n_val = RIAK_TRUE;
75  putmsg.n_val = options->n_val;
76  }
77  if (options->has_pw) {
78  putmsg.has_pw = RIAK_TRUE;
79  putmsg.pw = options->pw;
80  }
81  if (options->has_return_body) {
82  putmsg.has_return_body = RIAK_TRUE;
83  putmsg.return_body = options->return_body;
84  }
85  if (options->has_return_head) {
86  putmsg.has_return_head = RIAK_TRUE;
87  putmsg.return_head = options->return_head;
88  }
89  if (options->has_sloppy_quorum) {
90  putmsg.has_sloppy_quorum = RIAK_TRUE;
91  putmsg.sloppy_quorum = options->sloppy_quorum;
92  }
93  if (options->has_timeout) {
94  putmsg.has_timeout = RIAK_TRUE;
95  putmsg.timeout = options->timeout;
96  }
97  if (options->has_vclock) {
98  putmsg.has_vclock = RIAK_TRUE;
99  riak_binary_copy_to_pb(&(putmsg.vclock), options->vclock);
100  }
101  if (options->has_w) {
102  putmsg.has_w = RIAK_TRUE;
103  putmsg.w = options->w;
104  }
105  }
106 
107  riak_uint32_t msglen = rpb_put_req__get_packed_size (&putmsg);
108  riak_uint8_t* msgbuf = (riak_uint8_t*)(cfg->malloc_fn)(msglen);
109  if (msgbuf == NULL) {
110  return ERIAK_OUT_OF_MEMORY;
111  }
112  rpb_put_req__pack (&putmsg, msgbuf);
113 
114  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBPUTREQ, msglen, msgbuf);
115  if (request == NULL) {
116  return ERIAK_OUT_OF_MEMORY;
117  }
118  *req = request;
119  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_put_response_decode);
120 
121  return ERIAK_OK;
122 }

133 riak_error
134 riak_put_response_decode(riak_operation *rop,
135  riak_pb_message *pbresp,
136  riak_put_response **resp,
137  riak_boolean_t *done) {
138  // decode the PB response etc
139  riak_config *cfg = riak_operation_get_config(rop);
140  RpbPutResp *rpbresp = rpb_put_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
141  if (rpbresp == NULL) {
142  return ERIAK_MESSAGE_FORMAT;
143  }
144  *done = RIAK_TRUE;
145  riak_connection *cxn = riak_operation_get_connection(rop);
146  riak_log_debug(cxn, "riak_decode_put_response len=%d/pb unpack = 0x%lx\n", pbresp->len, (long)(rpbresp));
147  if (rpbresp == NULL) {
148  return ERIAK_OUT_OF_MEMORY;
149  }
150  int i = 0;
151  riak_put_response *response = (riak_put_response*)(cfg->malloc_fn)(sizeof(riak_put_response));
152  if (response == NULL) {
153  return ERIAK_OUT_OF_MEMORY;
154  }
155  memset(response, '\0', sizeof(riak_put_response));
156  response->_internal = rpbresp;
157  if (rpbresp->has_vclock) {
158  response->has_vclock = RIAK_TRUE;
159  response->vclock = riak_binary_copy_from_pb(cfg, &(rpbresp->vclock));
160  if (response->vclock == NULL) {
161  return ERIAK_OUT_OF_MEMORY;
162  }
163  }
164  if (rpbresp->has_key) {
165  response->has_key = RIAK_TRUE;
166  response->key = riak_binary_copy_from_pb(cfg, &(rpbresp->key));
167  if (response->key == NULL) {
168  return ERIAK_OUT_OF_MEMORY;
169  }
170  }
171  if (rpbresp->n_content > 0) {
172  riak_error err = riak_object_new_array(cfg, &(response->content), rpbresp->n_content);
173  if (err != ERIAK_OK) {
174  riak_free(cfg, &response);
175  return err;
176  }
177  response->n_content = rpbresp->n_content;
178  for(i = 0; i < rpbresp->n_content; i++) {
179  err = riak_object_new_from_pb(cfg, &(response->content[i]), rpbresp->content[i]);
180  // If any object allocation fails, clean up all previously allocated ones
181  if (err != ERIAK_OK) {
182  riak_object_free_array(cfg, &(response->content), i);
183  riak_free(cfg, &response);
184  return err;
185  }
186  }
187  }
188  *resp = response;
189 
190  return ERIAK_OK;
191 }

193 void
194 riak_put_response_print(riak_put_response *response,
195  char *target,
196  riak_size_t len) {
197 
198  char buffer[1024];
199  riak_int32_t wrote;
200  riak_int32_t left_to_write = len;
201  if ((left_to_write > 0) && (response->has_vclock)) {
202  riak_binary_hex_print(response->vclock, buffer, sizeof(buffer));
203  wrote = snprintf(target, left_to_write, "V-Clock: %s\n", buffer);
204  left_to_write -= wrote;
205  target += wrote;
206  }
207  if ((left_to_write > 0) && (response->has_key)) {
208  riak_binary_print(response->key, buffer, sizeof(buffer));
209  wrote = snprintf(target, left_to_write, "Key: %s\n", buffer);
210  left_to_write -= wrote;
211  target += wrote;
212  }
213  if (left_to_write > 0) {
214  wrote = snprintf(target, left_to_write, "Objects: %d\n", response->n_content);
215  left_to_write -= wrote;
216  target += wrote;
217  }
218  riak_uint32_t i;
219  for(i = 0; (i < response->n_content) && (left_to_write > 0); i++) {
220  wrote = riak_object_print(response->content[i], target, left_to_write);
221  }
222 }
