147 void
148 riak_event_callback(riak_bufferevent *bev,
149  short events,
150  void *ptr)
151 {
152  riak_event **rev_target = (riak_event**)ptr;
153  riak_event *rev = *rev_target;
154  if (events & BEV_EVENT_CONNECTED) {
155  riak_log_debug(rev, "%s","Connect okay.");
156  } else if (events & (BEV_EVENT_ERROR|BEV_EVENT_EOF)) {
157  char *reason = "BEV_EVENT_ERROR";
158  if (events & BEV_EVENT_ERROR) {
159  reason = "BEV_EVENT_EOF";
160  int err = bufferevent_socket_get_dns_error(bev);
161  if (err)
162  riak_log_error(rev, "DNS error: %s", evutil_gai_strerror(err));
163  }
164 #ifdef _RIAK_DEBUG
165  struct evbuffer *ev_read = bufferevent_get_input(bev);
166  struct evbuffer *ev_write = bufferevent_get_output(bev);
167 #endif
168  riak_log_debug(rev, "Closing because of %s [read event=%p, write event=%p]",
169  reason, (void*)ev_read, (void*)ev_write);
170  bufferevent_free(bev);
171  event_base_loopexit(rev->base, NULL);
172  } else if (events & BEV_EVENT_TIMEOUT) {
173  riak_log_debug(rev, "%s","Timeout Event");
174  bufferevent_free(bev);
175  event_base_loopexit(rev->base, NULL);
176  } else {
177  riak_log_debug(rev, "Event %d", events);
178  }
179 }

195 // MAIN RESULT CALLBACK
196 void
197 riak_read_result_callback(riak_bufferevent *bev,
198  void *ptr) {
199  riak_event **rev_target = (riak_event**)ptr;
200  riak_event *rev = *rev_target;
201  riak_context *ctx = (riak_context*)(rev->context);
202  riak_boolean_t done_streaming = RIAK_FALSE;
203  riak_size_t buflen;
204  while(RIAK_TRUE) {
205  // Are we in the middle of a message already?
206  if (rev->msglen_complete == RIAK_FALSE) {
207  // Read the first 32-bits which are the message size
208  // However, if a few size bytes were included during the last read, add them in
209  riak_uint32_t inmsglen = rev->msglen;
210  riak_size_t remaining_msg_len = sizeof(inmsglen) - rev->position;
211  riak_uint8_t *target = (riak_uint8_t*)(&inmsglen);
212  target += rev->position;
213  buflen = bufferevent_read(bev, target, remaining_msg_len);
214  target = (riak_uint8_t*)(&inmsglen);
215  // If we can't ready any more bytes, stop trying
216  if (buflen != remaining_msg_len) {
217  riak_log_debug(rev, "Expected %d bytes but received bytes = %d", remaining_msg_len, buflen);
218  if (buflen == 0) break;
219  // A few message size bytes of next message were in this buffer
220  // Stuff the partial size into rev->msglen and note the position
221  if (buflen < sizeof(inmsglen)) {
222  rev->position = buflen;
223  rev->msglen = inmsglen;
224  return;
225  }
226  abort(); // Something is hosed here
227  }
228 
229  rev->msglen_complete = RIAK_TRUE;
230  rev->msglen = ntohl(inmsglen);
231  riak_log_debug(rev, "%s","Read msglen = %d", rev->msglen);
232 
233  // TODO: Need to malloc new buffer each time?
234  rev->msgbuf = (riak_uint8_t*)(ctx->malloc_fn)(rev->msglen);
235  if (rev->msgbuf == NULL) {
236  riak_log_debug(rev, "%s", "Could not allocate buffer in riak_read_result_callback");
237  abort();
238  }
239  } else {
240  riak_log_debug(rev, "%s", "Continuation of partial message");
241  }
242 
243  riak_uint8_t *current_position = rev->msgbuf;
244  current_position += rev->position;
245  buflen = bufferevent_read(bev, (void*)current_position, rev->msglen - rev->position);
246  riak_log_debug(rev, "%s","read %d bytes at position %d, msglen = %d", buflen, rev->position, rev->msglen);
247  rev->position += buflen;
248  // Are we done yet? If not, break out and wait for the next callback
249  if (rev->position < rev->msglen) {
250  riak_log_debug(rev, "%s","Partial message received");
251  return;
252  }
253  assert(rev->position == rev->msglen);
254 
255  riak_uint8_t msgid = (rev->msgbuf)[0];
256  riak_pb_message *pbresp = riak_pb_message_new(ctx, msgid, rev->msglen, rev->msgbuf);
257  riak_error result;
258  rev->position = 0; // Reset on success
259  rev->msglen = 0;
260  rev->msglen_complete = RIAK_FALSE;
261 
262  // Response varies by data type
263  riak_error_response *err_response = NULL;
264  // Assume we are doing a single loop, unless told otherwise
265  done_streaming = RIAK_TRUE;
266  if (rev->decoder == NULL) {
267  riak_log_debug(rev, "%d NOT IMPLEMENTED", msgid);
268  abort();
269  }
270  if (msgid == MSG_RPBERRORRESP) {
271  result = riak_decode_error_response(rev, pbresp, &err_response, &done_streaming);
272  // Convert error response to a null-terminated string
273  char errmsg[2048];
274  riak_binary_print(err_response->errmsg, errmsg, sizeof(errmsg));
275  riak_log_debug(rev, "ERR #%d - %s\n", err_response->errcode, errmsg);
276  if (rev->error_cb) (rev->error_cb)(err_response, rev->cb_data);
277  exit(1);
278  }
279  // Decode the message from Protocol Buffers
280  result = (rev->decoder)(rev, pbresp, &(rev->response), &done_streaming);
281  riak_free(ctx, &pbresp);
282  riak_free(ctx, &rev->msgbuf);
283  // Call the user-defined callback for this message, when finished
284  if (done_streaming) {
285  if (rev->response_cb) (rev->response_cb)(rev->response, rev->cb_data);
286  }
287 
288  // Something is amiss
289  if (result)
290  abort();
291  }
292 
293  // What has been queued up
294  fflush(stdout);
295  event_base_dump_events(rev->base, stdout);
296 
297  if (done_streaming) {
298  riak_event_free(rev_target);
299  }
300 }
